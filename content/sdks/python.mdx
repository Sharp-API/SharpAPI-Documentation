## REST API

```python
import requests
from typing import Optional, List

class SharpApiClient:
    BASE_URL = 'https://api.sharpapi.io/api/v1'

    def __init__(self, api_key: str):
        self.api_key = api_key
        self.session = requests.Session()
        self.session.headers['X-API-Key'] = api_key

    def get_odds(self, sports: Optional[List[str]] = None) -> dict:
        params = {}
        if sports:
            params['sport'] = ','.join(sports)

        response = self.session.get(f'{self.BASE_URL}/odds', params=params)
        response.raise_for_status()
        return response.json()

    def get_ev(self, threshold: float = 2.0, books: Optional[List[str]] = None) -> dict:
        params = {'min_ev': threshold}
        if books:
            params['sportsbook'] = ','.join(books)

        response = self.session.get(f'{self.BASE_URL}/opportunities/ev', params=params)
        response.raise_for_status()
        return response.json()

    def get_arbitrage(self, min_profit: float = 0.5) -> dict:
        params = {'min_profit': min_profit}
        response = self.session.get(f'{self.BASE_URL}/opportunities/arbitrage', params=params)
        response.raise_for_status()
        return response.json()
```

## SSE Streaming

```python
import json
import sseclient
import requests
from typing import Optional, List

def stream_odds(api_key: str, sports: Optional[List[str]] = None):
    url = 'https://api.sharpapi.io/api/v1/stream'
    params = {'channel': 'all'}
    if sports:
        params['sport'] = ','.join(sports)
    headers = {'X-API-Key': api_key}

    response = requests.get(url, params=params, headers=headers, stream=True)
    client = sseclient.SSEClient(response)

    # Local state — odds:update events are DELTAS, not full snapshots
    odds_map = {}  # keyed by odds line ID
    is_ready = False

    for event in client.events():
        data = json.loads(event.data) if event.data else {}

        if event.event == 'connected':
            reconnected = data.get('reconnected', False)
            if reconnected:
                odds_map.clear()  # Clear stale state on reconnect
            print(f"Connected: stream {data['stream_id']}")

        elif event.event == 'snapshot':
            # Merge snapshot chunks into local state
            for key, value in data.items():
                if isinstance(value, list) and key not in ('ev', 'arbitrage', 'middles', 'low_hold'):
                    for odds in value:
                        odds_map[odds['id']] = odds

        elif event.event == 'snapshot:complete':
            is_ready = True
            print(f"Ready: {data['total_odds']} odds from {', '.join(data['books'])}")

        elif event.event == 'odds:update':
            # DELTA — merge changed odds into local state
            for book, odds_list in data.items():
                if not isinstance(odds_list, list):
                    continue
                for odds in odds_list:
                    odds_map[odds['id']] = odds

        elif event.event == 'odds:removed':
            # Delete removed odds from local state
            for odds_id in data.get('ids', []):
                odds_map.pop(odds_id, None)

        elif event.event == 'ev:detected':
            for opp in data:
                if not opp.get('possibly_stale', False):
                    print(f"+EV: {opp['selection']} {opp['ev_percent']}% on {opp['sportsbook']}")

        elif event.event == 'ev:expired':
            print(f"EV expired: {len(data['expired'])} opportunities")

        elif event.event == 'arb:detected':
            for arb in data:
                if not arb.get('possibly_stale', False):
                    print(f"Arb: {arb['profit_percent']}% — {arb['event_name']}")

        elif event.event == 'arb:expired':
            print(f"Arb expired: {len(data['expired'])} opportunities")

        elif event.event == 'heartbeat':
            pass  # silent keepalive


# Usage
stream_odds('YOUR_KEY', sports=['basketball'])
```

<Callout type="warning">
**Critical:** `odds:update` events are **deltas** — they only contain odds that changed. Your client must maintain local state (like `odds_map` above) and merge updates into it. Treating each event as a full snapshot will produce incorrect data.
</Callout>
